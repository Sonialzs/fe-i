## 85. 以下代码会输出什么

```js:index.js
const person = {
	name: 'Lydia',
	age: 21,
};

const changeAge = (x = { ...person }) => (x.age += 1);
const changeAgeAndName = (x = { ...person }) => {
	x.age += 1;
	x.name = 'Sarah';
};

changeAge(person);
changeAgeAndName();

console.log(person);
```
